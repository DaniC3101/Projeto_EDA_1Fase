94/*
95 * Use 8 bit buffer for IR timing in 50 ticks units.
96 * It is save to use 8 bit if RECORD_GAP_TICKS < 256, since any value greater 255 is interpreted as frame gap of 12750 us.
97 * The default for frame gap is currently 8000!
98 * But if we assume that for most protocols the frame gap is way greater than the biggest mark or space duration,
99 * we can choose to use a 8 bit buffer even for frame gaps up to 200000 us.
100 * This enables the use of 8 bit buffer even for more some protocols like B&O or LG air conditioner etc.
101 */
102#if RECORD_GAP_TICKS <= 400 // Corresponds to RECORD_GAP_MICROS of 200000. A value of 255 is foolproof, but we assume, that the frame gap is way greater than the biggest mark or space duration.
103typedef uint8_t IRRawbufType; // all timings up to the gap fit into 8 bit.
104#else
105typedef uint16_t IRRawbufType; // The gap does not fit into 8 bit ticks value. This must not be a reason to use 16 bit for buffer, but it is at least save.
106#endif

405/*
406 * Pulse parms are ((X*50)-MARK_EXCESS_MICROS) for the Mark and ((X*50)+MARK_EXCESS_MICROS) for the Space.
407 * First MARK is the one after the long gap
408 * Pulse parameters in microseconds
409 */
410#if !defined(TOLERANCE_FOR_DECODERS_MARK_OR_SPACE_MATCHING_PERCENT)
411#define TOLERANCE_FOR_DECODERS_MARK_OR_SPACE_MATCHING_PERCENT 25 // Relative tolerance (in percent) for matchTicks(), matchMark() and matchSpace() functions used for protocol decoding.
412#endif
413
414#define TICKS(us) ((us)/MICROS_PER_TICK) // (us)/50
415#if MICROS_PER_TICK == 50 && TOLERANCE_FOR_DECODERS_MARK_OR_SPACE_MATCHING_PERCENT == 25 // Defaults
416#define TICKS_LOW(us) ((us)/67 ) // 67 = MICROS_PER_TICK / ((100-25)/100) = (MICROS_PER_TICK * 100) / (100-25)
417#define TICKS_HIGH(us) (((us)/40) + 1) // 40 = MICROS_PER_TICK / ((100+25)/100) = (MICROS_PER_TICK * 100) / (100+25)
418#else
420//#define LTOL (1.0 - (TOLERANCE/100.))
421#define LTOL (100 - TOLERANCE_FOR_DECODERS_MARK_OR_SPACE_MATCHING_PERCENT)
423//#define UTOL (1.0 + (TOLERANCE/100.))
424#define UTOL (100 + TOLERANCE_FOR_DECODERS_MARK_OR_SPACE_MATCHING_PERCENT)
425#define TICKS_LOW(us) ((uint16_t ) ((long) (us) * LTOL / (MICROS_PER_TICK * 100) ))
426#define TICKS_HIGH(us) ((uint16_t ) ((long) (us) * UTOL / (MICROS_PER_TICK * 100) + 1))
427#endif
