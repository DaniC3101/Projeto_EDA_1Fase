102/**********************************************************************************************************************
103 * Interrupt Service Routine - Called every 50 us
104 *
105 * Duration in ticks of 50 us of alternating SPACE, MARK are recorded in irparams.rawbuf array.
106 * 'rawlen' counts the number of entries recorded so far.
107 * First entry is the SPACE between transmissions.
108 *
109 * As soon as one SPACE entry gets longer than RECORD_GAP_TICKS, state switches to STOP (frame received). Timing of SPACE continues.
110 * A call of resume() switches from STOP to IDLE.
111 * As soon as first MARK arrives in IDLE, gap width is recorded and new logging starts.
112 *
113 * With digitalRead and Feedback LED
114 * 15 pushs, 1 in, 1 eor before start of code = 2 us @16MHz + * 7.2 us computation time (6us idle time) + * pop + reti = 2.25 us @16MHz => 10.3 to 11.5 us @16MHz
115 * With portInputRegister and mask and Feedback LED code commented
116 * 9 pushs, 1 in, 1 eor before start of code = 1.25 us @16MHz + * 2.25 us computation time + * pop + reti = 1.5 us @16MHz => 5 us @16MHz
117 * => Minimal CPU frequency is 4 MHz
118 *
119 **********************************************************************************************************************/
120#if defined(ESP8266) || defined(ESP32)
121#pragma GCC diagnostic push
122#pragma GCC diagnostic ignored "-Wvolatile"
123IRAM_ATTR
124#endif
125void IRReceiveTimerInterruptHandler() {
126#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
127 digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
128#endif
129// 7 - 8.5 us for ISR body (without pushes and pops) for ATmega328 @16MHz
130
131#if defined(TIMER_REQUIRES_RESET_INTR_PENDING)
132 timerResetInterruptPending(); // reset TickCounterForISR interrupt flag if required (currently only for Teensy and ATmega4809)
133#endif
134
135// Read if IR Receiver -> SPACE [xmt LED off] or a MARK [xmt LED on]
136#if defined(__AVR__)
137 uint8_t tIRInputLevel = *irparams.IRReceivePinPortInputRegister & irparams.IRReceivePinMask;
138#else
139 uint_fast8_t tIRInputLevel = (uint_fast8_t) digitalReadFast(irparams.IRReceivePin);
140#endif
141
142 /*
143 * Increase TickCounter and clip it at maximum 0xFFFF / 3.2 seconds at 50 us ticks
144 */
145 if (irparams.TickCounterForISR < UINT16_MAX) {
146 irparams.TickCounterForISR++; // One more 50uS tick
147 }
148
149 /*
150 * Due to a ESP32 compiler bug https://github.com/espressif/esp-idf/issues/1552 no switch statements are possible for ESP32
151 * So we change the code to if / else if
152 */
153// switch (irparams.StateForISR) {
154//
155 if (irparams.StateForISR == IR_REC_STATE_IDLE) {
156 /*
157 * Here we are just resumed and maybe in the middle of a transmission
158 */
159 if (tIRInputLevel == INPUT_MARK) {
160 // check if we did not start in the middle of a transmission by checking the minimum length of leading space
161 if (irparams.TickCounterForISR > RECORD_GAP_TICKS) {
162#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
163// digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
164#endif
165 /*
166 * Gap between two transmissions just ended; Record gap duration + start recording transmission
167 * Initialize all state machine variables
168 */
169 irparams.OverflowFlag = false;
170// irparams.rawbuf[0] = irparams.TickCounterForISR;
171 irparams.initialGapTicks = irparams.TickCounterForISR; // Enabling 8 bit buffer since 4.4
172 irparams.rawlen = 1;
173 irparams.StateForISR = IR_REC_STATE_MARK;
174 } // otherwise stay in idle state
175 irparams.TickCounterForISR = 0; // reset counter in both cases
176 }
177
178 } else if (irparams.StateForISR == IR_REC_STATE_MARK) { // Timing mark
179 if (tIRInputLevel != INPUT_MARK) {
180 /*
181 * Mark ended here. Record mark time in rawbuf array
182 */
183#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
184// digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
185#endif
186 irparams.rawbuf[irparams.rawlen++] = irparams.TickCounterForISR; // record mark
187 irparams.StateForISR = IR_REC_STATE_SPACE;
188 irparams.TickCounterForISR = 0; // This resets the tick counter also at end of frame :-)
189 }
190
191 } else if (irparams.StateForISR == IR_REC_STATE_SPACE) { // Timing space
192 if (tIRInputLevel == INPUT_MARK) {
193 /*
194 * Space ended here. Check for overflow and record space time in rawbuf array
195 */
196 if (irparams.rawlen >= RAW_BUFFER_LENGTH) {
197 // Flag up a read OverflowFlag; Stop the state machine
198 irparams.OverflowFlag = true;
199 irparams.StateForISR = IR_REC_STATE_STOP;
200#if !defined(IR_REMOTE_DISABLE_RECEIVE_COMPLETE_CALLBACK)
201 /*
202 * Call callback if registered (not NULL)
203 */
204 if (irparams.ReceiveCompleteCallbackFunction != NULL) {
205 irparams.ReceiveCompleteCallbackFunction();
206 }
207#endif
208 } else {
209#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
210// digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
211#endif
212 irparams.rawbuf[irparams.rawlen++] = irparams.TickCounterForISR; // record space
213 irparams.StateForISR = IR_REC_STATE_MARK;
214 }
215 irparams.TickCounterForISR = 0;
216
217 } else if (irparams.TickCounterForISR > RECORD_GAP_TICKS) {
218 /*
219 * Maximum space duration reached here.
220 * Current code is ready for processing!
221 * We received a long space, which indicates gap between codes.
222 * Switch to IR_REC_STATE_STOP
223 * Don't reset TickCounterForISR; keep counting width of next leading space
224 */
225 /*
226 * These 2 variables allow to call resume() directly after decode.
227 * After resume(), decodedIRData.rawDataPtr->initialGapTicks and decodedIRData.rawDataPtr->rawlen are
228 * the first variables, which are overwritten by the next received frame.
229 * since 4.3.0.
230 * For backward compatibility, there are the same 2 statements in decode() if IrReceiver is not used.
231 */
232 IrReceiver.decodedIRData.initialGapTicks = irparams.initialGapTicks;
233 IrReceiver.decodedIRData.rawlen = irparams.rawlen;
234
235 irparams.StateForISR = IR_REC_STATE_STOP; // This signals the decode(), that a complete frame was received
236#if !defined(IR_REMOTE_DISABLE_RECEIVE_COMPLETE_CALLBACK)
237 /*
238 * Call callback if registered (not NULL)
239 */
240 if (irparams.ReceiveCompleteCallbackFunction != NULL) {
241 irparams.ReceiveCompleteCallbackFunction();
242 }
243#endif
244 }
245 } else if (irparams.StateForISR == IR_REC_STATE_STOP) {
246 /*
247 * Complete command received
248 * stay here until resume() is called, which switches state to IR_REC_STATE_IDLE
249 */
250#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
251// digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
252#endif
253 if (tIRInputLevel == INPUT_MARK) {
254 // Reset gap TickCounterForISR, to prepare for detection if we are in the middle of a transmission after call of resume()
255 irparams.TickCounterForISR = 0;
256 }
257 }
258
259#if !defined(NO_LED_FEEDBACK_CODE)
260 if (FeedbackLEDControl.LedFeedbackEnabled & LED_FEEDBACK_ENABLED_FOR_RECEIVE) {
261 setFeedbackLED(tIRInputLevel == INPUT_MARK);
262 }
263#endif
264
265#ifdef _IR_MEASURE_TIMING
266 digitalWriteFast(_IR_TIMING_TEST_PIN, LOW); // 2 clock cycles
267#endif
268
269}

318void IRrecv::setReceivePin(uint_fast8_t aReceivePinNumber) {
319 irparams.IRReceivePin = aReceivePinNumber;
320#if defined(__AVR__)
321# if defined(__digitalPinToBit)
322 if (__builtin_constant_p(aReceivePinNumber)) {
323 irparams.IRReceivePinMask = 1UL << (__digitalPinToBit(aReceivePinNumber));

335 irparams.IRReceivePinPortInputRegister = __digitalPinToPINReg(aReceivePinNumber);

338 }
339# else

341# endif
342#endif

381 timerConfigForReceive(); // no interrupts enabled here!
382 // Timer interrupt is enabled after state machine reset
383 if (sMicrosAtLastStopTimer != 0) {
384 irparams.TickCounterForISR += (micros() - sMicrosAtLastStopTimer) / MICROS_PER_TICK; // adjust TickCounterForISR for correct gap value, which is used for repeat detection

387 timerEnableReceiveInterrupt(); // Enables the receive sample timer interrupt which consumes a small amount of CPU every 50 us.

390#endif

700/**********************************************************************************************************************
701 * Common decode functions
702 **********************************************************************************************************************/
724bool IRrecv::decodePulseDistanceWidthData(uint_fast8_t aNumberOfBits, IRRawlenType aStartOffset, uint16_t aOneMarkMicros,
725 uint16_t aOneSpaceMicros, uint16_t aZeroMarkMicros, bool aMSBfirst) {
726
727 auto *tRawBufPointer = &decodedIRData.rawDataPtr->rawbuf[aStartOffset];
728
729 bool isPulseDistanceProtocol = (aOneMarkMicros == aZeroMarkMicros); // If true, we check aOneSpaceMicros -> pulse distance protocol
730
731 IRRawDataType tDecodedData = 0; // For MSB first tDecodedData is shifted left each loop
732 IRRawDataType tMask = 1UL; // Mask is only used for LSB first
733
734 for (uint_fast8_t i = aNumberOfBits; i > 0; i--) {
735 // get one mark and space pair
736 unsigned int tMarkTicks;
737 unsigned int tSpaceTicks;
738 bool tBitValue;
739
740 if (isPulseDistanceProtocol) {
741 /*
742 * PULSE_DISTANCE -including PULSE_DISTANCE_WIDTH- here.
743 * !!!We only check variable length space indicating a 1 or 0!!!
744 */
745 tRawBufPointer++;
746 tSpaceTicks = *tRawBufPointer++; // maybe buffer overflow for last bit, but we do not evaluate this value :-)
747 tBitValue = matchSpace(tSpaceTicks, aOneSpaceMicros); // Check for variable length space indicating a 1 or 0
748
749 } else {
750 /*
751 * PULSE_WIDTH here.
752 * !!!We only check variable length mark indicating a 1 or 0!!!
753 */
754 tMarkTicks = *tRawBufPointer++;
755 tBitValue = matchMark(tMarkTicks, aOneMarkMicros); // Check for variable length mark indicating a 1 or 0
756 tRawBufPointer++;
757 }
758
759 if (aMSBfirst) {
760 tDecodedData <<= 1;
761 }
762
763 if (tBitValue) {
764 // It's a 1 -> set the bit
765 if (aMSBfirst) {
766 tDecodedData |= 1;
767 } else {
768 tDecodedData |= tMask;
769 }
770 IR_TRACE_PRINTLN(F("=> 1"));
771 } else {
772 // do not set the bit
773 IR_TRACE_PRINTLN(F("=> 0"));
774 }
775 tMask <<= 1;
776 }

1014uint_fast8_t IRrecv::getBiphaselevel() {
1015 uint_fast8_t tLevelOfCurrentInterval; // 0 (SPACE) or 1 (MARK)
1016
1017 if (sBiphaseDecodeRawbuffOffset >= decodedIRData.rawlen) {
1018 return SPACE; // After end of recorded buffer, assume space.
1019 }
1020
1021 tLevelOfCurrentInterval = (sBiphaseDecodeRawbuffOffset) & 1; // on odd rawbuf offsets we have mark timings
1022
1023 /*
1024 * Setup data if sUsedTimingIntervals is 0
1025 */
1026 if (sBiphaseUsedTimingIntervals == 0) {
1027 uint16_t tCurrentTimingWith = decodedIRData.rawDataPtr->rawbuf[sBiphaseDecodeRawbuffOffset];
1028 uint16_t tMarkExcessCorrection = (tLevelOfCurrentInterval == MARK) ? MARK_EXCESS_MICROS : -MARK_EXCESS_MICROS;
1029
1030 if (matchTicks(tCurrentTimingWith, sBiphaseTimeUnit + tMarkExcessCorrection)) {
1031 sBiphaseCurrentTimingIntervals = 1;
1032 } else if (matchTicks(tCurrentTimingWith, (2 * sBiphaseTimeUnit) + tMarkExcessCorrection)) {
1033 sBiphaseCurrentTimingIntervals = 2;
1034 } else if (matchTicks(tCurrentTimingWith, (3 * sBiphaseTimeUnit) + tMarkExcessCorrection)) {
1035 sBiphaseCurrentTimingIntervals = 3;
1036 } else {
1037 return -1;
1038 }
1039 }
1040
1041// We use another interval from tCurrentTimingIntervals
1042 sBiphaseUsedTimingIntervals++;
1043
1044// keep track of current timing offset
1045 if (sBiphaseUsedTimingIntervals >= sBiphaseCurrentTimingIntervals) {
1046 // we have used all intervals of current timing, switch to next timing value
1047 sBiphaseUsedTimingIntervals = 0;
1048 sBiphaseDecodeRawbuffOffset++;
1049 }
1050
1051 IR_TRACE_PRINTLN(tLevelOfCurrentInterval);
1052
1053 return tLevelOfCurrentInterval;

1094bool IRrecv::decodeHash() {
1095 unsigned long hash = FNV_BASIS_32; // the result is the same no matter if we use a long or unsigned long variable
1096
1097// Require at least 6 samples to prevent triggering on noise
1098 if (decodedIRData.rawlen < 6) {
1099 IR_DEBUG_PRINT(F("HASH: "));
1100 IR_DEBUG_PRINT(F("Data length="));
1101 IR_DEBUG_PRINT(decodedIRData.rawlen);
1102 IR_DEBUG_PRINTLN(F(" is less than 6"));
1103 return false;
1104 }
1105 for (IRRawlenType i = 1; (i + 2) < decodedIRData.rawlen; i++) {
1106 // Compare mark with mark and space with space
1107 uint_fast8_t value = compare(decodedIRData.rawDataPtr->rawbuf[i], decodedIRData.rawDataPtr->rawbuf[i + 2]);
1108 // Add value into the hash
1109 hash = (hash * FNV_PRIME_32) ^ value;
1110 }
1111
1112 decodedIRData.decodedRawData = hash;
1113 decodedIRData.numberOfBits = 32;
1114 decodedIRData.protocol = UNKNOWN;
1115
1116 return true;
1117}
