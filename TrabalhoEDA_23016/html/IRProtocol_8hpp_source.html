133 size_t tLength = p->print(static_cast<uint32_t>(value >> 32), base);

163void printIRResultShort(Print *aSerial, IRData *aIRDataPtr, bool aPrintRepeatGap) {
164 if (aIRDataPtr->flags & IRDATA_FLAGS_WAS_OVERFLOW) {
165 aSerial->println(F("Overflow"));
166 return;
167 }
168 aSerial->print(F("Protocol="));
169 aSerial->print(getProtocolString(aIRDataPtr->protocol));
170 if (aIRDataPtr->protocol == UNKNOWN) {
171#if defined(DECODE_HASH)
172 aSerial->print(F(" Hash=0x"));
173#if (__INT_WIDTH__ < 32)
174 aSerial->print(aIRDataPtr->decodedRawData, HEX);
175#else
176 PrintULL::print(aSerial,aIRDataPtr->decodedRawData, HEX);
177#endif
178
179#endif
180#if !defined(DISABLE_CODE_FOR_RECEIVER)
181 aSerial->print(' ');
182 aSerial->print((aIRDataPtr->rawlen + 1) / 2, DEC);
183 aSerial->println(F(" bits (incl. gap and start) received"));
184#endif
185 } else {
186#if defined(DECODE_DISTANCE_WIDTH)
187 if (aIRDataPtr->protocol != PULSE_DISTANCE && aIRDataPtr->protocol != PULSE_WIDTH) {
188#endif
189 /*
190 * New decoders have address and command
191 */
192 aSerial->print(F(" Address=0x"));
193 aSerial->print(aIRDataPtr->address, HEX);
194
195 aSerial->print(F(" Command=0x"));
196 aSerial->print(aIRDataPtr->command, HEX);
197
198 if (aIRDataPtr->flags & IRDATA_FLAGS_EXTRA_INFO) {
199 aSerial->print(F(" Extra=0x"));
200 aSerial->print(aIRDataPtr->extra, HEX);
201 }
202
203 if (aIRDataPtr->flags & IRDATA_FLAGS_PARITY_FAILED) {
204 aSerial->print(F(" Parity fail"));
205 }
206
207 if (aIRDataPtr->flags & IRDATA_FLAGS_TOGGLE_BIT) {
208 aSerial->print(F(" Toggle=1"));
209 }
210#if defined(DECODE_DISTANCE_WIDTH)
211 }
212#endif
213 if (aIRDataPtr->flags & (IRDATA_FLAGS_IS_AUTO_REPEAT | IRDATA_FLAGS_IS_REPEAT)) {
214 aSerial->print(' ');
215 if (aIRDataPtr->flags & IRDATA_FLAGS_IS_AUTO_REPEAT) {
216 aSerial->print(F("Auto-"));
217 }
218 aSerial->print(F("Repeat"));
219#if !defined(DISABLE_CODE_FOR_RECEIVER)
220 if (aPrintRepeatGap) {
221 aSerial->print(F(" gap="));
222 aSerial->print((uint32_t) aIRDataPtr->initialGapTicks * MICROS_PER_TICK);
223 aSerial->print(F("us"));
224 }
225#else
226 (void)aPrintRepeatGap;
227#endif
228 }
229
230 /*
231 * Print raw data
232 */
233 if (!(aIRDataPtr->flags & IRDATA_FLAGS_IS_REPEAT) || aIRDataPtr->decodedRawData != 0) {
234 aSerial->print(F(" Raw-Data=0x"));
235#if (__INT_WIDTH__ < 32)
236 aSerial->print(aIRDataPtr->decodedRawData, HEX);
237#else
238 PrintULL::print(aSerial, aIRDataPtr->decodedRawData, HEX);
239#endif
240 /*
241 * Print number of bits processed
242 */
243 aSerial->print(' ');
244 aSerial->print(aIRDataPtr->numberOfBits, DEC);
245 aSerial->print(F(" bits"));
246
247 if (aIRDataPtr->flags & IRDATA_FLAGS_IS_MSB_FIRST) {
248 aSerial->println(F(" MSB first"));
249 } else {
250 aSerial->println(F(" LSB first"));
251 }
252
253 } else {
254 aSerial->println();
255 }
256 }
257}

259/**********************************************************************************************************************
260 * Function to bit reverse OLD MSB values of e.g. NEC.
261 **********************************************************************************************************************/
262uint8_t bitreverseOneByte(uint8_t aValue) {
263// uint8_t tReversedValue;
264// return __builtin_avr_insert_bits(0x01234567, aValue, tReversedValue);
265// 76543210
266 aValue = (aValue >> 4) | (aValue << 4); // Swap in groups of 4
267// 32107654
268 aValue = ((aValue & 0xcc) >> 2) | ((aValue & 0x33) << 2); // Swap in groups of 2
269// 10325476
270 aValue = ((aValue & 0xaa) >> 1) | ((aValue & 0x55) << 1); // Swap bit pairs
271// 01234567
272 return aValue;
273}

275uint32_t bitreverse32Bit(uint32_t aInput) {
276// __builtin_avr_insert_bits();
277 LongUnion tValue;
278 tValue.UByte.HighByte = bitreverseOneByte(aInput);
279 tValue.UByte.MidHighByte = bitreverseOneByte(aInput >> 8);
280 tValue.UByte.MidLowByte = bitreverseOneByte(aInput >> 16);
281 tValue.UByte.LowByte = bitreverseOneByte(aInput >> 24);
282 return tValue.ULong;
283}
